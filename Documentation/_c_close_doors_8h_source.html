48  // Needed informations queries succeeded ?

53  {
54  // Above close speed limit ?
55  if (this->mVehicleSpeed.GetCurrentValue() >= C_CLOSE_SPEED_THRESHOLD)
56  {
57  // We reached close speed limit : we can now open again if needed
58  this->mShouldOpen = true ;
59 
60  // At least one door opened ?
61  if (!this->mDriverDoorStatus.GetCurrentValue() || !this->mPassengerDoorStatus.GetCurrentValue())
62  {
63  // The system should close the doors ?
64  if (this->mShouldClose)
65  {
66  // Try to close the doors. Succeeded ?
67  if (this->mDoorsLock.CloseDoors(this->mCAN))
68  {
69  // Don't need to close again
70  this->mShouldClose = false ;
71  }
72  }
73  }
74  else
75  {
76  // Don't need to close again
77  this->mShouldClose = false ;
78  }
79  }
80  // Under close speed limit
81  else
82  {
83  // We are under close speed limit : we can now close again if needed
84  this->mShouldClose = true ;
85 
86  // At least one door closed ?
87  if (this->mDriverDoorStatus.GetCurrentValue() || this->mPassengerDoorStatus.GetCurrentValue())
88  {
89  // The system should open the doors ?
90  if (this->mShouldOpen)
91  {
92  // Ignition off ?
93  if (this->mElectricalPower.GetCurrentValue() == FRS::EElecPowerStatus::EPS_OFF)
94  {
95  // Open the doors succeeded ?
96  if (this->mDoorsLock.OpenDoors(this->mCAN))
97  {
98  // Don't need to open again
99  this->mShouldOpen = false ;
100  }
101  }
102  }
103  }
104  else
105  {
106  // Don't need to open again
107  this->mShouldOpen = false ;
108  }
109  }
110  }
