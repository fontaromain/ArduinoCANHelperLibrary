64  {

66  bool lDisplayStatus = ((pFrame.GetData()[5] & 0x04) != 0x0) ;
67  bool lTripStatus = ((pFrame.GetData()[3] & 0x80) != 0x0) ;
68 
69  // Return correct state
70  if (lDisplayStatus && lTripStatus)
71  {
72  return ECombiBtnStatus::S_BOTH_PRESSED ;
73  }
74  else if (lDisplayStatus && !lTripStatus)
75  {
76  return ECombiBtnStatus::S_DISPLAY_PRESSED ;
77  }
78  else if (!lDisplayStatus && lTripStatus)
79  {
80  return ECombiBtnStatus::S_TRIP_PRESSED ;
81  }
82  else
83  {
84  return ECombiBtnStatus::S_NONE_PRESSED ;
85  }
