80  static ECombiBtnStatus OnParse(const ACANFrame& pFrame)
81  {
82  // Extract both statuses
83  bool lDisplayStatus = ((pFrame.GetData()[5] & 0x04) != 0x0) ;
84  bool lTripStatus = ((pFrame.GetData()[3] & 0x80) != 0x0) ;
85 
86  // Return correct state
87  if (lDisplayStatus && lTripStatus)
88  {
89  return ECombiBtnStatus::S_BOTH_PRESSED ;
90  }
91  else if (lDisplayStatus && !lTripStatus)
92  {
93  return ECombiBtnStatus::S_DISPLAY_PRESSED ;
94  }
95  else if (!lDisplayStatus && lTripStatus)
96  {
97  return ECombiBtnStatus::S_TRIP_PRESSED ;
98  }
99  else
100  {
101  return ECombiBtnStatus::S_NONE_PRESSED ;
102  }
103  }
