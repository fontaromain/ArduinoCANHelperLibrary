71  bool ReadFrom(ICANConnector& pCAN)
72  {
73  // A message available ?
74  if (pCAN.WaitHasMessages())
75  {
76  // Read succeed ?
77  if (pCAN.Read(this->mTempAddress, this->mTempLength, this->mTempData))
78  {
79  // Did we just read a single frame ?
80  if (CANIsSingleFrame(this->mTempData[0]))
81  {
82  // For each data received in temp buffer
83  for (int lDataIndex = 0 ; lDataIndex < this->mTempLength ; lDataIndex++)
84  {
85  // Copy data over correct buffer
86  this->mData[lDataIndex] = this->mTempData[lDataIndex] ;
87  }
88 
89  // And get correct length and address
90  this->mLength = this->mTempLength ;
91  this->mAddress = this->mTempAddress ;
92 
93  // Nothing more to do, it's perfect !
94  return true ;
95  }
96  // A first frame ?
97  else if (CANIsFirstFrame(this->mTempData[0]))
98  {
99  // Extract data from the first frame
100  {
101  // Extract the lenght and store the address
102  this->mLength = (((unsigned short)(this->mTempData[0] & 0x0F)) << 8) + this->mTempData[1] ;
103  this->mAddress = this->mTempAddress ;
104 
105  // For each data received in temp buffer skipping first byte (first frame ID)
106  for (int lDataIndex = 1 ; lDataIndex < this->mTempLength ; lDataIndex++)
107  {
108  // Copy data over correct buffer
109  this->mData[lDataIndex - 1] = this->mTempData[lDataIndex] ;
110  }
111  }
112 
113  // We now have to ask CAN bus to send us the remaining data
114  // Store current index position to append correctly next replies
115  unsigned short lCurrentIndex = 4 ; // First frame is 4 data (8 - 2 (First frame / lenght ID) - 2 (Mode and Pid))
116 
117  // While we haven't read all the data
118  while (lCurrentIndex < this->mLength)
119  {
120  // Make and send flow control succeeded ?
121  if (this->MakeAndSendFlowControlTempFrame(this->mAddress, pCAN))
122  {
123  // Wait for at least a reply
124  if (pCAN.WaitHasMessages())
125  {
126  // And now, parse the consecutive frames we might have received
127  pCAN.Read(this->mTempAddress, this->mTempLength, this->mTempData) ;
128 
129  // A consecutive frame ?
130  if (CANIsConsecutiveFrame(this->mTempData[0]))
131  {
132  // For each data received in temp buffer skipping first byte (consecutive frame ID)
133  for (int lDataIndex = 1 ; lDataIndex < this->mTempLength ; lDataIndex++)
134  {
135  // Copy data over correct buffer
136  this->mData[lDataIndex - 1 + lCurrentIndex] = this->mTempData[lDataIndex] ;
137  }
138 
139  // Increment current index with read data
140  lCurrentIndex += 7 ; // Consecutive frame is 7 data (8 - consecutive frame ID)
141  }
142  // Error : not a consecutive frame !
143  else
144  {
145  // Clean
146  this->Clean() ;
147 
148  // Leave
149  return false ;
150  }
151  }
152  // Reply timeout
153  else
154  {
155  return false ;
156  }
157  }
158  // Send timeout
159  else
160  {
161  return false ;
162  }
163  }
164  #ifdef CREADCANFRAME_DEBUG_RECEIVE
165  // Print debug
166  this->Print() ;
167  #endif
168 
169  // Seems ok
170  return true ;
171  }
172  // Unknow message
173  else
174  {
175  // For each data received in temp buffer
176  for (int lDataIndex = 0 ; lDataIndex < this->mTempLength ; lDataIndex++)
177  {
178  // Copy data over correct buffer
179  this->mData[lDataIndex] = this->mTempData[lDataIndex] ;
180  }
181 
182  // And get correct length and address
183  this->mLength = this->mTempLength ;
184  this->mAddress = this->mTempAddress ;
185 
186  #ifdef CREADCANFRAME_DEBUG_RECEIVE
187  // Print debug
188  PRINT_STR("Received a specific frame type which is not standard :") ;
189  PRINTLNHEX(this->mData[0] & 0xF0) ;
190  #endif
191 
192  // Specific frame type which is non standard
193  return true ;
194  }
195  }
196  else
197  {
198  // Read failed
199  return false ;
200  }
201  }
202  else
203  {
204  // No messages
205  return false ;
206  }
207  }

225  bool MakeAndSendFlowControlTempFrame(unsigned long pOriginalAddress, ICANConnector& pCAN)
226  {
227  // Define address and length
228  this->mTempAddress = pOriginalAddress - CAN_FLOW_CONTROL_ADDRESS_VALUE ;
229  this->mTempLength = CAN_FRAME_LENGHT ;
230 
231  // Define data
232  this->mTempData[0] = CAN_FLOW_FRAME_ID << 4 ; // Flow control identifier
233  this->mTempData[1] = 0x01 ; // We want the CAN bus to sends us one consecutive frame only
234  this->mTempData[2] = CAN_NO_DATA ;
235  this->mTempData[3] = CAN_NO_DATA ;
236  this->mTempData[4] = CAN_NO_DATA ;
237  this->mTempData[5] = CAN_NO_DATA ;
238  this->mTempData[6] = CAN_NO_DATA ;
239  this->mTempData[7] = CAN_NO_DATA ;
240 
241  // Send our flow request using the correct address and wait until send works or time out
242  return pCAN.WaitSend(this->mAddress - CAN_FLOW_CONTROL_ADDRESS_VALUE, CAN_FRAME_LENGHT, this->mTempData) ;
243  }
