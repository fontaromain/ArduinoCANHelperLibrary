321  static bool InitializeCAN(ICANConnector& pCAN, SetupFiltersAndMasks pSetupFunc = nullptr)
322  {
323  // Declare everything we need
324  char lTestCounter = 0 ;
325  unsigned long lCANId = 0 ;
326  unsigned char lLength = 0 ;
327  unsigned char lData[8] = { 0, 0, 0, 0, 0, 0, 0, 0 } ;
328 
329  // While CAN initialization fails
330  while (!pCAN.Initialize(pSetupFunc))
331  {
332  // More than 20 tries ?
333  if (lTestCounter++ > 20)
334  {
335  // Failed !
336  return false ;
337  }
338 
339  // Wait some time before trying again
340  DELAY(100) ;
341  }
342 
343  // Unstack all possible message we might have received even if filters have been set accordingly
344  while (pCAN.HasMessages())
345  {
346  pCAN.Read(lCANId, lLength, lData) ;
347  }
348 
349  // Valid if no errors so far !
350  return !pCAN.HasError() ;
351  }
